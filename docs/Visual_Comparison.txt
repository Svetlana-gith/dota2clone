│  │   border-left-width: 3px;                                           │   │
│  │   border-color: rgba(255, 107, 107, 1.0);                          │   │

│  │   font-size: 15px;                                                  │   │

│  │ }                                                                   │   │

│                                                                             │
│  ❌ Нельзя переиспользовать                                                 │
│  ❌ Дублирование для success/warning                                        │
│  ❌ Сложно менять цвета                                                     │
└─────────────────────────────────────────────────────────────────────────────┘

┌─────────────────────────────────────────────────────────────────────────────┐
│  TAILWIND-STYLE APPROACH                                                    │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                             │
│  C++:                                                                       │
│  ┌─────────────────────────────────────────────────────────────────────┐   │
│  │ auto label = std::make_shared<CLabel>("Error", "ErrorLabel");      │   │
│  │ label->AddClass("bg-error");                                        │   │
│  │ label->AddClass("border-l-3");                                      │   │
│  │ label->AddClass("border-error");                                    │   │
│  │ label->AddClass("rounded");                                         │   │
│  │ label->AddClass("text-error");                                      │   │
│  │ label->AddClass("px-3");                                            │   │
│  │ container->AddChild(label);                                         │   │
│  └─────────────────────────────────────────────────────────────────────┘   │
│                                                                             │
│  CSS (16 lines total for error + success + warning):                       │
│  ┌─────────────────────────────────────────────────────────────────────┐   │
│  │ /* Positioning only */                                              │   │
│  │ #ErrorLabel { width: 87%; height: 7%; x: 6%; y: 49%; }             │   │
│  │                                                                     │   │
│  │ /* Utility classes (reusable) */                                   │   │
│  │ .bg-error { background-color: rgba(255, 107, 107, 0.15); }         │   │
│  │ .border-l-3 { border-left-width: 3px; }                            │   │
│  │ .border-error { border-color: rgba(255, 107, 107, 1.0); }          │   │
│  │ .rounded { border-radius: 6px; }                                   │   │
│  │ .text-error { color: rgba(255, 107, 107, 1.0); }                   │   │
│  │ .px-3 { padding-left: 12px; padding-right: 12px; }                 │   │
│  └─────────────────────────────────────────────────────────────────────┘   │
│                                                                             │
│  ✅ Переиспользуемые классы                                                 │
│  ✅ Success/warning за 1 минуту                                             │
│  ✅ Консистентные цвета/размеры                                             │
└─────────────────────────────────────────────────────────────────────────────┘
